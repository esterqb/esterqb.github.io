    for i in range(2, n):

    return suma

def es_par(num):
return num % 2 == 0

def numeros_pares(lista):
return [n for n in lista if es_par(n)]

def obtener_lista():
lista = []
while True:
    entrada = input("Dame n√∫meros (Termina con 'fin'): ")
    if entrada.lower() == "fin":
        break
    try:
        numero = int(entrada)
        lista.append(numero)
    except ValueError:
        print("Entrada no v√°lida. Introduce un n√∫mero o 'fin'.")
return lista

# Prueba interactiva
lista_usuario = obtener_lista()
print(f"La lista original es: {lista_usuario}")
print("N√∫meros pares:", numeros_pares(lista_usuario))

# Prueba directa (sin input)
