while True:
    entrada = input("Ingresa un número entero positivo: ")
    if entrada.isdigit():
        numero = int(entrada)
        if numero > 0:
            print(f"Número válido ingresado: {numero}")
            break
        else:
            print("El número debe ser mayor que cero.")
    else:
        print("Entrada inválida. Debes ingresar un número entero positivo.")


Ejercicio 3

palabras = input("Escribe varias palabras separadas por espacios: ").split()
conteo = {}

for palabra in palabras:
    conteo[palabra] = conteo.get(palabra, 0) + 1

print("\nOcurrencias de cada palabra:")
for palabra, cantidad in conteo.items():
    print(f"{palabra}: {cantidad}")


Ejercicio 4

notas = {}
print("Ingresa las notas de hasta 5 estudiantes:")

for i in range(5):
    nombre = input(f"Nombre del estudiante {i + 1}: ")
    while True:
        nota_input = input(f"Nota de {nombre}: ")
        try:
            nota = float(nota_input)
            notas[nombre] = nota
            break
        except ValueError:
            print("Nota inválida. Ingresa un número.")

print("\nNotas de los estudiantes:")
for estudiante, nota in notas.items():
    print(f"{estudiante}: {nota}")

promedio = sum(notas.values()) / len(notas)
print(f"\nPromedio de notas: {promedio:.2f}")


Ejercicio 5

ingredientes_disponibles = ["queso", "tomate", "jamón", "champiñones", "aceitunas", "cebolla", "piña", "pimiento"]
seleccionados = []

print("Ingredientes disponibles:")
print(", ".join(ingredientes_disponibles))

while len(seleccionados) < 5:
    ingr = input(f"Selecciona un ingrediente ({len(seleccionados)+1}/5): ").lower()
    if ingr in ingredientes_disponibles and ingr not in seleccionados:
        seleccionados.append(ingr)
    elif ingr in seleccionados:
        print("Ya seleccionaste ese ingrediente.")
    else:
        print("Ingrediente no disponible.")

print("\nIngredientes seleccionados para tu pizza:")
print(", ".join(seleccionados))


Ejercicio 6

personas = []
print("Ingresa nombres de personas. Escribe 'fin' para terminar:")

while True:
    nombre = input("Nombre: ")
    if nombre.lower() == "fin":
        break
    personas.append(nombre)

buscar = input("\n¿A quién deseas buscar?: ")

if buscar in personas:
    posicion = personas.index(buscar)
    print(f"{buscar} está en la lista en la posición {posicion}.")
else:
    print(f"{buscar} no se encontró en la lista.")
