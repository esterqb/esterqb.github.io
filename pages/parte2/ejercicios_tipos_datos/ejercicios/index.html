# Ejercicio14.
cantidad = float(input("Introduzca la cantidad a invertir: "))
interes = float(input("Introduzca el porcentaje de interés anual: "))
anios = int(input("Introduzca el número de años: "))
#En estas tres líneas anteriores se pide por consola que se introduzcan los valores que pide el ejercicio y se almacenen
#en float (ya que puede no ser un número entero) menos la variable años que siempre será un entero.
print("\nAño\tCapital obtenido") #se imprimirán a la izquierda el número de años (hacia abajo) y en otra columan el capital
for anios in range(1, anios + 1): #se inicia un bucle for que irá desde el primer año hasta el último año que se haya introducido
    capital = cantidad * (1 + interes / 100) ** anios #se declara la variable capital para saber cuál es la cantidad después de recibir el interes cada X años introducidos por consola.
    print(anios, "\t", capital) #se imprime el año (1º,2º... hasta llegar al introducido, y en otra columna -tabulado- la cantidad con el interés añadido)
