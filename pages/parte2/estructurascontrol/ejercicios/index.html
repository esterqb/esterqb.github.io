#Ejercicio 11.
# Solicita los extremos del rango
inicio = int(input("Primer número: "))
fin = int(input("Segundo número: "))
print(f"Números primos entre {inicio} y {fin}:")
# Función para verificar si un número es primo
def es_primo(n):
    if n < 2:
        return False
    for i in range(2, int(n**0.5) + 1):
        if n % i == 0:
            return False
    return True
# Recorre el rango y muestra los números primos
for numm in range(inicio, fin + 1):
    if es_primo(numm):
        print(numm, end=", ")

#Ejercicio 13.
import random
# Genera un número aleatorio entre 1 y 100
numero_secreto = random.randint(1, 100)
# Bucle hasta que el usuario adivine correctamente
while True:
    intento = int(input("Adivina el número (entre 1 y 100): "))
    if intento < numero_secreto:
        print("Es mayor...")
    elif intento > numero_secreto:
        print("Es menor...")
    else:
        print(f"¡Correcto! El número era {numero_secreto}")
        break

#Ejercicio 14.
num7 = int(input("Cantidad de números a mostrar: "))
fibonacci = []
a, b = 0, 1
for _ in range(num7):
    fibonacci.append(a)
    a, b = b, a + b
print("Serie de Fibonacci:", ", ".join(map(str, fibonacci)))
