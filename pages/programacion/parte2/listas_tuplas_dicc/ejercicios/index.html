def filtrar_pares(lista):
return [num for num in lista if num % 2 == 0]


listas_util.py

import random

def eliminar_duplicados(lista):
    return list(set(lista))

def generar_lista_aleatoria(tamano, minimo=1, maximo=100):
    return [random.randint(minimo, maximo) for _ in range(tamano)]

def encontrar_maximo(lista):
    return max(lista)

def encontrar_minimo(lista):
    return min(lista)


tuplas_util.py

def crear_lista_tuplas(lista):
    return list(enumerate(lista))

def contar_elementos_tupla(tupla, elemento):
    return tupla.count(elemento)


diccionarios_util.py

def crear_diccionario(claves, valores):
    return dict(zip(claves, valores))

def contar_ocurrencias_lista(lista):
    conteo = {}
    for num in lista:
        conteo[num] = conteo.get(num, 0) + 1
    return conteo

def revertir_diccionario(diccionario):
    return {v: k for k, v in diccionario.items()}


texto_util.py

def contar_palabras(texto):
    palabras = texto.split()
    conteo = {}
    for palabra in palabras:
        conteo[palabra] = conteo.get(palabra, 0) + 1
    return conteo
